#include <Arduino.h>
#include <Servo.h>


#define BUTTON 2

Servo myServo;

int bef = 1;
int aft = 1;
int anzahl = 0;
int buttonFlag = false;
int pressTime = 0;
int array[] = {};
void setup()
{
  Serial.begin(9600); // The baudrate of Serial monitor is set in 9600
  pinMode(BUTTON, INPUT_PULLUP);
  myServo.attach(14);//PIN 5
}

void loop()
{
    bef = digitalRead(BUTTON);
    if((bef != aft) && (bef == 0)) {
        buttonFlag = true;
        pressTime = millis();
    }
    if(buttonFlag) {
        if(millis()-pressTime < 15) {
            if(digitalRead(BUTTON) == 1) {
                buttonFlag = false;
            }
        } else
        {
            Serial.println("Schuss");
            anzahl++;
            Serial.print("Anzahl:  ");
            Serial.println(anzahl);
        }
    }
    aft = bef;
   //Serial.println(digitalRead(BUTTON));
}
